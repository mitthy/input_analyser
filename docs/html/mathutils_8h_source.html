47  template<typename Function, typename NumberType>
48  input_data_t integral(NumberType first, NumberType last, Function fx, const std::true_type&) {
49  NumberType max = std::max(first, last);
50  NumberType min = std::min(first, last);
51  NumberType step = static_cast<NumberType>(0.05);
52  NumberType _first = first, _second = first + step;
53  input_data_t sum = 0;
54  int iterations = (max - _second) / step;
55  //Uses the sum of areas of trapezia to calculate the integral of fx.
56  //(y0 + y1)/2 * (x1 - x0) + (y1 + y2)/2 * (x2 - x1)...
57  //Since (x(n) - x(n-1)) delta x is always equal to step, we don't need to calculate it.
58  #pragma omp parallel for reduction (+:sum)
59  for(int i = 0; i < iterations; ++i) {
60  //This is necessary for OpenMP support.
61  NumberType x_first = _first + (i * step);
62  NumberType x_second = _second + (i * step);
63  NumberType y_first = fx(x_first), y_second = fx(x_second);
64  sum += static_cast<input_data_t>(y_first + y_second) / 2.0f * step;
65  }
66  return sum;
67  }

79  input_data_t integral(NumberType first, NumberType last, Function fx, const std::false_type&) {
80  NumberType max = std::max(first, last);
81  NumberType min = std::min(first, last);
82  NumberType step = static_cast<NumberType>(1);
83  NumberType _first = first, _second = _first + step;
84  int iterations = (max - _second);
85  input_data_t sum = 0;
86  //Uses the sum of areas of trapezia to calculate the integral of fx.
87  //(y0 + y1)/2 * (x1 - x0) + (y1 + y2)/2 * (x2 - x1)...
88  //Since (x(n) - x(n-1)) delta x is always equal to step, we don't need to calculate it.
89  #pragma omp parallel for reduction (+:sum)
90  for(int i = 0; i < iterations; ++i) {
91  //This is necessary for OpenMP support.
92  NumberType x_first = _first + i;
93  NumberType x_second = _second + i;
94  NumberType y_first = fx(_first), y_second = fx(_second);
95  sum += static_cast<input_data_t>(y_first + y_second) / 2.0f;
96  _first += step;
97  _second += step;
98  }
