214 MonteCarlo::MonteCarlo(Iterator begin, Iterator end, std::size_t classes): _data_count(std::distance(begin, end)) {
215  //If the difference between the max element and the minimum element is lesser than this value, we don't split the data into classes.
216  input_data_t EPSLON = 0;
217  input_data_t sqrt_sz = std::sqrt(_data_count);
218  if(classes) {
219  EPSLON = std::nextafter(std::numeric_limits<input_data_t>::min(), (input_data_t)1.0) * classes * 4;
220  }
221  else {
222  EPSLON = std::nextafter(std::numeric_limits<input_data_t>::min(), (input_data_t)1.0) * sqrt_sz * 4;
223  }
224  auto min_max = std::minmax_element(begin, end);
225  input_data_t min = *min_max.first, max = *min_max.second;
226  //We split the input data into classes. For now, it just gets the minimum between the amount of data / 10 + 1 and 15.
227  int number_of_classes = classes;
228  if(!number_of_classes || number_of_classes > _data_count) {
229  number_of_classes = max - min < EPSLON ? 1 : sqrt_sz;
230  }
231  input_data_t step = 0;
232  //The step that we take between each classes.
233  if(number_of_classes > 1) {
234  step = (max - min) / (number_of_classes - 1);
235  }
236  //Calculates the bound for each class.
237  input_data_t lower_bound = step ? min - step / 2 : min;
238  input_data_t upper_bound = step ? min + step / 2 : max;
239  //Insert all the classes into a vector.
240  for(int i = 0; i < number_of_classes; ++i) {
241  input_data_t avg = lower_bound + (upper_bound - lower_bound) / 2;
242  monte_carlo_class new_class = {avg, 0, 0, lower_bound, upper_bound};
243  lower_bound += step;
244  upper_bound += step;
245  _organized_data.push_back(new_class);
246  }
247  //For each element, we do a binary search to find its position in the class vector and then we increment the amount of data of that class.
248  while(begin != end) {
249  std::size_t low = 0, high = _organized_data.size();
250  input_data_t val = *begin++;
251  //Binary search to find first element less than val.
252  while(low != high) {
253  //We do this to avoid overflow.
254  std::size_t mid = (low >> 1) + (high >> 1) + ((low & high & 1));
255  if(val < _organized_data[mid].lower_bound) {
256  high = mid;
257  }
258  else {
259  low = mid + 1;
260  }
261  }
262  ++_organized_data[--low].class_count;
263  }
264  input_data_t acum = 0;
265  for(monte_carlo_class& klass : _organized_data) {
266  klass.acum_probability = acum;
267  acum += klass.class_count * static_cast<input_data_t>(1.0) / _data_count;
268  }
269 }
