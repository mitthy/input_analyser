210 MonteCarlo::MonteCarlo(Iterator begin, Iterator end): _data_count(std::distance(begin, end)) {
211  //If the difference between the max element and the minimum element is lesser than this value, we don't split the data into classes.
212  const float EPSLON = 1e-4;
213  auto min_max = std::minmax_element(begin, end);
214  float min = *min_max.first, max = *min_max.second;
215  //We split the input data into classes. For now, it just gets the minimum between the amount of data / 10 + 1 and 15.
216  int number_of_classes = max - min < EPSLON ? 1 : std::min((_data_count / 10) + 1, (size_t)15);
217  float step = 0;
218  //The step that we take between each classes.
219  if(number_of_classes > 1) {
220  step = (max - min) / (number_of_classes - 1);
221  }
222  //Calculates the bound for each class.
223  float lower_bound = step ? min - step / 2 : min;
224  float upper_bound = step ? min + step / 2 : max;
225  //Insert all the classes into a vector.
226  for(int i = 0; i < number_of_classes; ++i) {
227  float avg = lower_bound + (upper_bound - lower_bound) / 2;
228  monte_carlo_class new_class = {avg, 0, 0, lower_bound, upper_bound};
229  lower_bound += step;
230  upper_bound += step;
231  _organized_data.push_back(new_class);
232  }
233  //For each element, we do a binary search to find its position in the class vector and then we increment the amount of data of that class.
234  while(begin != end) {
235  std::size_t low = 0, high = _organized_data.size();
236  float val = *begin++;
237  //Binary search to find first element less than val.
238  while(low != high) {
239  //We do this to avoid overflow.
240  std::size_t mid = (low >> 1) + (high >> 1) + ((low & high & 1));
241  if(val < _organized_data[mid].lower_bound) {
242  high = mid;
243  }
244  else {
245  low = mid + 1;
246  }
247  }
248  ++_organized_data[--low].class_count;
249  }
250  float acum = 0;
251  for(monte_carlo_class& klass : _organized_data) {
252  klass.acum_probability = acum;
253  acum += klass.class_count * 1.0f / _data_count;
254  }
255 }
